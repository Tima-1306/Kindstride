    <h2 class="sec-h" style="margin-top:12px;" data-key="aid_h">What to do right now</h2>
    <p class="sec-sub" data-key="aid_sub">Select the condition. These are triage steps — not a substitute for veterinary care.</p>
    <div class="ebar"><i class="ti ti-alert-triangle ebar-icon"></i><div class="ebar-text"><strong data-key="aid_warn_h">First aid is not treatment.</strong><span data-key="aid_warn_b">Everything here is what you do while getting to professional help — not instead of it.</span></div></div>
    <div class="ctabs">
      <button class="ctab on" onclick="showCond('shock',this)"><i class="ti ti-heart-rate-monitor"></i> <span data-key="cond_shock">Shock</span></button>
      <button class="ctab" onclick="showCond('wound',this)"><i class="ti ti-bandage"></i> <span data-key="cond_wound">Open wound</span></button>
      <button class="ctab" onclick="showCond('dehy',this)"><i class="ti ti-droplet"></i> <span data-key="cond_dehy">Dehydration</span></button>
      <button class="ctab" onclick="showCond('heat',this)"><i class="ti ti-thermometer"></i> <span data-key="cond_heat">Heatstroke</span></button>
      <button class="ctab" onclick="showCond('poison',this)"><i class="ti ti-skull"></i> <span data-key="cond_poison">Poisoning</span></button>
      <button class="ctab" onclick="showCond('fracture',this)"><i class="ti ti-bone"></i> <span data-key="cond_frac">Fracture</span></button>

    <div id="cond-shock" class="cpanel on">
      <span class="sev crit"><i class="ti ti-alert-triangle"></i> <span data-key="sev_crit">Critical — vet immediately</span></span>
      <ul class="csteps">
        <li><strong data-key="sh1s">Warmth before anything else.</strong> <span data-key="sh1b">Wrap loosely in fleece. A warm water bottle in a sock; never direct heat.</span></li>
        <li><strong data-key="sh2s">Minimum handling.</strong> <span data-key="sh2b">Move once to a quiet warm space, then leave alone.</span></li>
        <li><strong data-key="sh3s">No food, no water.</strong> <span data-key="sh3b">The vet will administer fluids correctly.</span></li>
        <li><strong data-key="sh4s">Monitor breathing every 2 minutes.</strong> <span data-key="sh4b">Open-mouth breathing in cats = rapid deterioration.</span></li>
      </ul>
      <div class="cnote" data-key="sh_note">Signs of shock: pale/white gums, glassy eyes, rapid shallow breathing, cold extremities, unresponsiveness.</div>
    </div>
    <div id="cond-wound" class="cpanel">
      <span class="sev mod"><i class="ti ti-bandage"></i> <span data-key="sev_mod">Moderate — assess depth</span></span>
      <ul class="csteps">
        <li><strong data-key="wo1s">Stop bleeding first.</strong> <span data-key="wo1b">Firm pressure with clean cloth for 8–10 minutes. Do not lift to check.</span></li>
        <li><strong data-key="wo2s">No hydrogen peroxide, no alcohol.</strong> <span data-key="wo2b">Saline or clean water only.</span></li>
        <li><strong data-key="wo3s">Do not remove embedded objects.</strong> <span data-key="wo3b">Stabilize with padding around — not over — the object.</span></li>
        <li><strong data-key="wo4s">All bite wounds go to the vet.</strong> <span data-key="wo4b">The entry hole hides deep tissue damage.</span></li>
      </ul>
      <div class="cnote" data-key="wo_note">Chest, abdomen or neck wounds; bleeding after 10 minutes — emergency vet immediately.</div>
    </div>
    <div id="cond-dehy" class="cpanel">
      <span class="sev mod"><i class="ti ti-droplet"></i> <span data-key="sev_mod">Moderate — act within hours</span></span>
      <ul class="csteps">
        <li><strong data-key="dh1s">Test skin turgor.</strong> <span data-key="dh1b">Pinch back-of-neck skin and release. Slow return = dehydration.</span></li>
        <li><strong data-key="dh2s">Small sips frequently.</strong> <span data-key="dh2b">Not a large bowl — rapid drinking triggers vomiting.</span></li>
        <li><strong data-key="dh3s">Wet food better than dry.</strong> <span data-key="dh3b">Do not force-feed via syringe unless trained.</span></li>
        <li><strong data-key="dh4s">Sunken eyes or sticky gums = IV fluids needed.</strong> <span data-key="dh4b">Oral water alone is not fast enough.</span></li>
      </ul>
      <div class="cnote blue" data-key="dh_note">Young animals dehydrate critically fast — in under 2 hours. Treat with full urgency.</div>
    </div>
    <div id="cond-heat" class="cpanel">
      <span class="sev crit"><i class="ti ti-thermometer"></i> <span data-key="sev_crit">Critical — every minute matters</span></span>
      <ul class="csteps">
        <li><strong data-key="ht1s">Move from heat immediately.</strong> <span data-key="ht1b">Cool room, shade, or car with AC.</span></li>
        <li><strong data-key="ht2s">Cool water — not ice.</strong> <span data-key="ht2b">Apply to paw pads, armpits, neck. Ice constricts vessels, trapping core heat.</span></li>
        <li><strong data-key="ht3s">No wet towel covers.</strong> <span data-key="ht3b">Use flowing cool water or fan on damp fur.</span></li>
        <li><strong data-key="ht4s">Vet within 30 minutes even if recovered.</strong> <span data-key="ht4b">Internal organ damage is invisible.</span></li>
      </ul>
      <div class="cnote" data-key="ht_note">Flat-faced breeds (Persian, Bulldog, Pug) at extreme risk even in mild heat.</div>
    </div>
    <div id="cond-poison" class="cpanel">
      <span class="sev crit"><i class="ti ti-skull"></i> <span data-key="sev_crit">Critical — call vet before acting</span></span>
      <ul class="csteps">
        <li><strong data-key="po1s">Do not induce vomiting without vet instruction.</strong> <span data-key="po1b">Caustics cause more damage coming back up.</span></li>
        <li><strong data-key="po2s">Identify the substance.</strong> <span data-key="po2b">Bring the packaging — treatment depends entirely on the toxin.</span></li>
        <li><strong data-key="po3s">Cats: all lily species, paracetamol, essential oils, xylitol, antifreeze.</strong></li>
        <li><strong data-key="po4s">Dogs: grapes, raisins, xylitol, chocolate, rat poison, macadamia nuts.</strong></li>
      </ul>
      <div class="cnote" data-key="po_note">Cats lack liver enzymes to process many compounds safe for humans and dogs.</div>
    </div>
    <div id="cond-fracture" class="cpanel">
      <span class="sev mod"><i class="ti ti-bone"></i> <span data-key="sev_mod">Moderate to severe — immobilize and go</span></span>
      <ul class="csteps">
        <li><strong data-key="fr1s">Do not splint yourself.</strong> <span data-key="fr1b">Incorrect splinting causes nerve damage.</span></li>
        <li><strong data-key="fr2s">Support the limb as found.</strong> <span data-key="fr2b">Do not straighten. Move as one unit.</span></li>
        <li><strong data-key="fr3s">Open fractures are emergencies.</strong> <span data-key="fr3b">Bone visible: cover with damp clean cloth and go immediately.</span></li>
        <li><strong data-key="fr4s">Keep still and warm.</strong> <span data-key="fr4b">Stillness in darkness is the best pre-vet care.</span></li>
      </ul>
      <div class="cnote blue" data-key="fr_note">Pelvic fractures after road accidents need imaging — a visual check is not enough.</div>

function setLang(lang, el){
  currentLang = lang;
  document.querySelectorAll('.lang-btn').forEach(function(b){b.classList.remove('on');});
  el.classList.add('on');
  var t = LANG[lang];
  document.querySelectorAll('[data-key]').forEach(function(node){

    if(t[k] !== undefined) node.textContent = t[k];

  renderRescueSteps();
  buildAllBreedGrids();
}

    if(s.warn) html+='<div class="warn-n"><i class="ti ti-alert-circle"></i> '+esc(s.warn)+'</div>';
    if(s.blue) html+='<div class="blue-n"><i class="ti ti-info-circle"></i> '+esc(s.blue)+'</div>';
    html+='</div></div>';
  });
  document.getElementById('rescue-steps-render').innerHTML = html;
}

function buildBreedGrid(type){
  var grid=document.getElementById(type+'-grid');
  if(!grid) return;
  grid.innerHTML='';
  document.getElementById(type+'-detail').classList.remove('vis');
  var langBreeds = (SITE_DATA.breeds[currentLang] || SITE_DATA.breeds.en)[type];
  langBreeds.forEach(function(b,i){
    var d=document.createElement('div');
    d.className='bcard';
    d.innerHTML='<div class="bcard-emoji">'+b.emoji+'</div><div class="bcard-name">'+esc(b.name)+'</div><div class="bcard-trait">'+esc(b.trait)+'</div>';

  });
}

function showBreedDetail(type,i){
  var langBreeds = (SITE_DATA.breeds[currentLang] || SITE_DATA.breeds.en)[type];

  document.getElementById(type+'-title').textContent=b.name;
  document.getElementById(type+'-tag').textContent=b.tag;

  b.rows.forEach(function(r){
    html+='<div class="brow"><div class="brow-label">'+esc(r.l)+'</div>';

function openBreedModal(type,index){
  editingBreedType=type; editingBreedIndex=index;
  if(index===null){
    document.getElementById('modal-title').textContent=LANG[currentLang].modal_add;
    ['m-name','m-emoji','m-trait','m-tag','m-rows'].forEach(function(id){document.getElementById(id).value='';});
  } else {
    var b=SITE_DATA.breeds.en[type][index];
    document.getElementById('modal-title').textContent=b.name;
    document.getElementById('m-name').value=b.name;
    document.getElementById('m-emoji').value=b.emoji;
    document.getElementById('m-trait').value=b.trait;
    document.getElementById('m-tag').value=b.tag;
    document.getElementById('m-rows').value=b.rows.map(function(r){ return r.v!==null&&r.v!==undefined?r.l+' | '+r.v+' | '+r.n:r.l+' | '+r.n; }).join('\n');
  }
  document.getElementById('breed-modal').classList.add('open');
}

function closeBreedModal(){ document.getElementById('breed-modal').classList.remove('open'); }

function saveBreed(){
  var name=document.getElementById('m-name').value.trim();
  if(!name){alert(LANG[currentLang].modal_name+' ?'); return;}
  var rows=document.getElementById('m-rows').value.trim().split('\n').filter(function(l){return l.trim();}).map(function(line){
    var p=line.split('|').map(function(x){return x.trim();});
    if(p.length>=3&&!isNaN(p[1])) return {l:p[0],v:parseInt(p[1]),c:'g',n:p.slice(2).join(' | ')};
    if(p.length>=2) return {l:p[0],v:null,n:p.slice(1).join(' | ')};
    return {l:line,v:null,n:''};
  });
  var breed={name:name,emoji:document.getElementById('m-emoji').value.trim()||'🐾',trait:document.getElementById('m-trait').value.trim(),tag:document.getElementById('m-tag').value.trim(),rows:rows};
  if(editingBreedIndex===null) SITE_DATA.breeds.en[editingBreedType].push(breed);
  else SITE_DATA.breeds.en[editingBreedType][editingBreedIndex]=breed;
  saveData(); renderAdminBreedList(editingBreedType); buildBreedGrid(editingBreedType); updateBreedCount(); closeBreedModal(); showToast(LANG[currentLang].toast_saved);
}
